# Type Definitions

Core types and interfaces used in the hua framework.

## Config Types

### HuaConfig

```typescript
interface HuaConfig {
  preset?: Preset
  i18n?: {
    defaultLanguage: string
    supportedLanguages: string[]
    fallbackLanguage?: string
    namespaces?: string[]
    translationLoader?: 'api' | 'static' | 'custom'
    translationApiPath?: string
    loadTranslations?: (lang: string, namespace: string) => Promise<Record<string, string>>
    debug?: boolean
    autoUpdateHtmlLang?: boolean
    initialTranslations?: Record<string, Record<string, Record<string, string>>>
    languageStore?: unknown
    storageKey?: string
  }
  motion?: {
    style?: 'smooth' | 'dramatic' | 'minimal'
    defaultPreset?: 'product' | 'marketing'
    enableAnimations?: boolean
    duration?: number
    easing?: string
  }
  state?: { persist?: boolean; ssr?: boolean }
  fileStructure?: { enforce?: boolean }
  icons?: {
    set?: 'phosphor' | 'lucide' | 'iconsax'
    weight?: 'thin' | 'light' | 'regular' | 'bold' | 'duotone' | 'fill'
    size?: number
    color?: string
    strokeWidth?: number
  }
  toast?: {
    position?: 'top-right' | 'top-left' | 'bottom-right' | 'bottom-left' | 'top-center' | 'bottom-center'
    maxToasts?: number
  }
  branding?: {
    name?: string
    logo?: string
    colors?: {
      primary?: string; secondary?: string; accent?: string
      success?: string; warning?: string; error?: string; info?: string
    }
    typography?: {
      fontFamily?: string
      fontSize?: { h1?: string; h2?: string; h3?: string; body?: string }
    }
    customVariables?: Record<string, string>
  }
  plugins?: HuaPlugin[]
  license?: { apiKey?: string; type?: 'free' | 'pro' | 'enterprise' }
}
```

### Preset Types

```typescript
type PresetName = 'product' | 'marketing'

type PresetConfig = {
  type: PresetName
  motion?: { duration?: number; easing?: string }
  spacing?: { default?: 'sm' | 'md' | 'lg' | 'xl' }
}

type Preset = PresetName | PresetConfig
```

### HuaPageProps

```typescript
interface HuaPageProps {
  children: React.ReactNode
  title?: string
  description?: string
  vibe?: 'clean' | 'fancy' | 'minimal'
  i18nKey?: string
  enableMotion?: boolean          // default: true
  motion?: 'fadeIn' | 'slideUp' | 'slideLeft' | 'slideRight' | 'scaleIn' | 'bounceIn'
  seo?: {
    keywords?: string[]
    ogImage?: string
    ogTitle?: string
    ogDescription?: string
    ogType?: string
  }
  enableErrorBoundary?: boolean   // default: true
  errorBoundaryFallback?: React.ReactNode | ((error: Error, reset: () => void) => React.ReactNode)
}
```

### ErrorBoundaryProps

```typescript
interface ErrorBoundaryProps {
  children: React.ReactNode
  fallback?: React.ReactNode | ((error: Error, reset: () => void) => React.ReactNode)
  onError?: (error: Error, errorInfo: React.ErrorInfo) => void
  onReset?: () => void
}
```
